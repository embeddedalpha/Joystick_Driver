19void GPIO_Interrupt_Setup(int pin, int edge_select, uint32_t priority)
20{
21 // Enable interrupt mask register for the pin
22 EXTI ->IMR |= 1 << pin;
23
24 // Configure edge selection
25 switch (edge_select) {
26 case 0:
27 EXTI ->RTSR |= 1 << pin;
28 break;
29 case 1:
30 EXTI ->FTSR |= 1 << pin;
31 break;
32 case 2:
33 EXTI ->RTSR |= 1 << pin;
34 EXTI ->FTSR |= 1 << pin;
35 break;
36 }
37
38 // Configure NVIC based on pin number
39 // Configure NVIC based on pin number
40 IRQn_Type irq;
41 if (pin <= 4) {
42 irq = (IRQn_Type)(EXTI0_IRQn + pin);
43 } else if (pin <= 9) {
44 irq = EXTI9_5_IRQn;
45 } else {
46 irq = EXTI15_10_IRQn;
47 }
48 NVIC_SetPriority(irq, priority);
49 NVIC_EnableIRQ(irq);
50}

133void GPIO_Pin_Init(GPIO_TypeDef *Port, uint8_t pin, uint8_t mode, uint8_t output_type, uint8_t speed, uint8_t pull, uint8_t alternate_function)
134{
135 GPIO_Clock_Enable(Port);
136 Port -> MODER |= mode << (pin*2);
137
138 if(output_type != GPIO_Configuration.Output_Type.None) Port -> OTYPER |= output_type << pin;
139 if(speed != GPIO_Configuration.Speed.None) Port -> OSPEEDR |= speed << (pin*2);
140 if(pull != GPIO_Configuration.Pull.None) Port -> PUPDR |= pull << (pin*2);
141 if(pin < 8) Port -> AFR[0] |= alternate_function << (pin*4);
142 else Port -> AFR[1] |= alternate_function << ((pin-8)*4);
143}
